// PRICE var prevRate = null; function fetchPrice() { fetch('https://api.frankfurter.dev/v1/latest?base=EUR&symbols=USD') .then(function(r){ return r.json(); }) .then(function(d) { var rate = d.rates.USD; document.getElementById('price').textContent = rate.toFixed(4); if (prevRate !== null) { var diff = rate - prevRate; var pct = (diff / prevRate * 100); var el = document.getElementById('price-chg'); el.textContent = (diff >= 0 ? '▲ +' : '▼ ') + diff.toFixed(4) + ' (' + (diff>=0?'+':'') + pct.toFixed(2) + '%)'; el.style.color = diff >= 0 ? 'var(--green)' : '#ef4444'; } else { var el = document.getElementById('price-chg'); el.textContent = '● live ECB'; el.style.color = 'var(--green)'; } prevRate = rate; var now = new Date(); document.getElementById('p-time').textContent = 'обновлено ' + String(now.getHours()).padStart(2,'0') + ':' + String(now.getMinutes()).padStart(2,'0'); }).catch(function(){ document.getElementById('p-time').textContent = 'ошибка соединения'; }); } function fetchHistory() { var today = new Date(); var fmt = function(d){ return d.getFullYear()+'-'+String(d.getMonth()+1).padStart(2,'0')+'-'+String(d.getDate()).padStart(2,'0'); }; var d30 = new Date(today); d30.setDate(d30.getDate()-30); fetch('https://api.frankfurter.dev/v1/'+fmt(d30)+'..'+fmt(today)+'?base=EUR&symbols=USD') .then(function(r){ return r.json(); }) .then(function(d) { var rates = Object.values(d.rates).map(function(x){ return x.USD; }); var high = Math.max.apply(null, rates); var low = Math.min.apply(null, rates); var range = Math.round((high - low) * 10000); document.getElementById('p-open').textContent = rates[0].toFixed(4); document.getElementById('p-high').textContent = high.toFixed(4); document.getElementById('p-low').textContent = low.toFixed(4); document.getElementById('p-range').textContent = range + ' пп'; document.getElementById('vol-val').textContent = range + ' пп'; var level, color, pct; if (range < 150) { level='Низкая'; color='var(--green)'; pct=15; } else if (range < 300) { level='Умеренная'; color='#f59e0b'; pct=45; } else if (range < 500) { level='Высокая'; color='#ef4444'; pct=75; } else { level='Очень высокая'; color='#ef4444'; pct=92; } document.getElementById('vol-label').textContent = level; document.getElementById('vol-val').style.color = color; document.getElementById('vol-marker').style.left = pct + '%'; }).catch(function(){}); } fetchPrice(); fetchHistory(); setInterval(fetchPrice, 30000); setInterval(fetchHistory, 3600000);
